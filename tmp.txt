Table cast {
  movie_id     varchar [primary key, Ref: > movies.movie_id]
  person_id     varchar [primary key, Ref: > persons.person_id]
  character_id int64 [primary key, Ref: > characters.character_id]
  
}

Table characters {
  character_id int64 [primary key]
  name    varchar
}

Table directors {
  movie_id    varchar [primary key, Ref: > movies.movie_id]
  person_id    varchar [primary key, Ref: > persons.person_id]
}

Table movie_genres {
  movie_id varchar [primary key, Ref: - movies.movie_id]
  genre_id int64 [primary key, Ref: > genres.genre_id]
}

Table genres {
  genre_id varchar [primary key]
  genre    varchar
}

Table knownformovies {
  person_id    varchar [primary key, Ref: > persons.person_id]
  movie_id    varchar [primary key, Ref: - movies.movie_id]
}

Table movies {
    movie_id varchar [primary key]
    titleType          varchar
    primaryTitle       varchar
    originalTitle      varchar
    isAdult             int64
    startYear           int64
    endYear           float64
    runtimeMinutes    float64
}

Table persons {
    person_id varchar [primary key]
    primaryName     varchar
    birthYear      float64
    deathYear      float64
}
Table principals {
    movie_id varchar [primary key, Ref: > movies.movie_id]
    ordering int64 [primary key]
    person_id varchar [Ref: > persons.person_id]
    profession_id int64 [Ref: - professions.profession_id]
    job         varchar
}

Table person_profession {
  person_id varchar [primary key, Ref: - persons.person_id]
  profession_id int64 [primary key, Ref: > professions.profession_id]
}

Table professions {
    profession_id int64 [primary key]
    jobName    varchar
}

Table ratings {
    movie_id varchar [primary key, Ref: - movies.movie_id, null]
    averageRating    float64
    numVotes           int64
}

Table titles {
    movie_id varchar [primary key, Ref: > movies.movie_id]
    ordering int64 [primary key]
    title              varchar
    region             varchar
    language           varchar
    types              varchar
    attributes         varchar
    isOriginalTitle     int64
}

Table writers {
    movie_id    varchar [primary key, Ref: > movies.movie_id]
    person_id    varchar [primary key, Ref: - persons.person_id]
}
